def recv_protocol_ack(opts = {})
  nsock = opts[:sock] || sock
  data = safe_get_once(nsock)
  begin
    ack = Rex::Proto::Rmi::Model::ProtocolAck.decode(StringIO.new(data))
  rescue ::RuntimeError
    return nil
  end

  ack
end

def safe_get_once(nsock = sock)
  data = ''
  begin
    res = nsock.get_once
  rescue ::EOFError
    res = nil
  end

  until res.nil? || res.length < 1448
    data << res
    begin
      res = nsock.get_once
    rescue ::EOFError
      res = nil
    end
  end

  data << res if res
  data
end
